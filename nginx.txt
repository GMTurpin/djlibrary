La ruta de acceso a este archivo en el servidor es:
(libros) root@libgmt:~# cd /etc/nginx/sites-available



upstream library_app {
  server unix:/djangolib/libros/run/gunicorn.sock fail_timeout=0;
}

server {

    listen   80;
    server_name 178.128.34.140;

    access_log /djangolib/libros/logs/nginx-access.log;
    error_log /djangolib/libros/logs/nginx-error.log;

    location /static/ {
        alias   /djangolib/libros/djlibrary/staticfiles/;
    }


    location / {
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $http_host;
        proxy_redirect off;

        if (!-f $request_filename) {
            proxy_pass http://library_app;
            break;
        }
    }
}
